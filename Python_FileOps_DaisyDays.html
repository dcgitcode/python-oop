open() 函数 — 语法介绍
f = open(文件名, 模式)
# 示例：打开文件准备写入
f = open("hello.txt", "w")
文件名 要打开的文件路径，如 "score.txt"（字符串类型）
模式 决定打开后能做什么：读？写？追加？——详见下一页对比表
返回值 返回一个文件对象，用变量 f 接收，后续通过 f 来操作文件

open() 模式对比表

模式 含义 文件不存在时 文件已存在时 典型场景
"r" ≡ "rt" 只读（文本） ❌ 报错 FileNotFoundError 从头读取 读取配置文件
"w" ≡ "wt" 只写（文本） ✅ 自动新建 ⚠️ 清空原内容 宿管建晚归记录
"a" ≡ "at" 追加（文本） ✅ 自动新建 在末尾追加 班长追加班费账本
"rb" 只读（二进制） ❌ 报错 从头读取 读取图片/音频
"wb" 只写（二进制） ✅ 自动新建 ⚠️ 清空原内容 保存图片/音频

🔤 "t" = 文本模式（默认，可省略）→ 操作字符串 str   ｜  "b" = 二进制模式 → 操作字节 bytes
🎯 口诀："r" 必须文件存在，"w"/"a" 不存在自动新建；"w" 清空重写，"a" 末尾追加

安全文件操作：try / except / finally

# 有缺陷的代码：如果中间出错，close() 永远不执行！
f = open("test.txt", "r")       # 假设文件不存在...
# 读取操作...
f.close()                    # ⚠️ 如果上面报错，这行永远不执行！

# ✅ 正确做法：try/except/finally 保证文件一定关闭
# 口诀：打开文件放 try，出错进 except，关闭放 finally！
f = None
try:
    f = open("test.txt", "r")
    # 读取或写入操作...
except FileNotFoundError:
    print("错误：文件不存在！")
except PermissionError:
    print("错误：没有权限访问该文件！")
except Exception as e:
    print(f"发生未知错误：{e}")
finally:
    if f:          # f=None 初始化是防止 finally 里报 NameError
        f.close()
        print("文件已关闭")

第三节：写文本文件

"

我们已经学会了安全地打开和关闭文件。
现在问题来了——打开之后，怎么把内容写进去？

就像你打开了笔记本，总得知道怎么握笔写字吧？

📌 核心方法：f.write("内容")

⚠️ write() 不会自动换行，需要手动加 \n

⚠️ write() 只能写入字符串，数字需先 str() 转换

"w" 覆盖 vs "a" 追加 — 社团日志

❌ 错误：用 "w" 追加日志
# 先用"w"模式写入
f = None
try:
    f = open("log.txt", "w")
    f.write("第一次活动记录\n")
finally:
    if f: f.close()

# 第二天再用"w"写入
f = None
try:
    f = open("log.txt", "w")
    f.write("第二次活动记录\n")
finally:
    if f: f.close()
# 结果：第一次记录消失了！
✅ 正确：用 "a" 追加日志
# 社团负责人追加活动日志
f = None
try:
    f = open("社团日志.txt", "a")
    f.write("2025-09-10 招新活动\n")
    print("追加成功！")
except Exception as e:
    print(f"写入失败：{e}")
finally:
    if f:
        f.close()

# 结果：保留所有历史记录 ✓

🎯 口诀：新建/覆盖旧内容 → "w"（宿管晚归表）  ｜  保留旧内容追加 → "a"（学生会日志、班费账本）

write() 写入 — 代码示例

# write() 方法：一次写入一个字符串（教案案例）
f = None
try:
    f = open("score.txt", "w")
    f.write("姓名：张三\n")
    f.write("成绩：95\n")
    print("写入成功！")
except Exception as e:
    print(f"写入失败：{e}")
finally:
    if f:
        f.close()

# ⚠️ write() 不会自动换行，需要手动加 \n
# ⚠️ write() 只能写入字符串，数字会报 TypeError
# f.write(100)        # ❌ TypeError: write() argument must be str
f.write(str(100))     # ✅ 先转字符串

案例一：单个学生信息写入

# 案例：将一个学生的信息写入文件（含异常处理）
name   = "张三"
stu_id = "2024001"
phone  = "13800138000"

f = None
try:
    f = open("student_info.txt", "w")
    f.write(f"姓名：{name}\n")
    f.write(f"学号：{stu_id}\n")
    f.write(f"手机：{phone}\n")
    print("学生信息已保存！")
except Exception as e:
    print(f"保存失败：{e}")
finally:
    if f:
        f.close()
        print("文件已关闭")

案例二：批量学生信息写入

# 案例：将多个学生信息写入文件（含异常处理）
students = [
    {"name": "张三", "id": "2024001", "phone": "13800138000"},
    {"name": "李四", "id": "2024002", "phone": "13900139000"},
    {"name": "王五", "id": "2024003", "phone": "15000150000"},
    {"name": "赵六", "id": "2024004", "phone": "18600186000"},
    {"name": "孙七", "id": "2024005", "phone": "13700137000"},
]
f = None
try:
    f = open("students.txt", "w")
    f.write("学号      姓名   手机号\n")
    f.write("-" * 35 + "\n")
    for stu in students:
        line = stu["id"] + "  " + stu["name"] + "   " + stu["phone"] + "\n"
        f.write(line)
    print(f"已保存{len(students)}名学生信息到students.txt")
except Exception as e:
    print(f"保存失败：{e}")
finally:
    if f:
        f.close()
        print("文件已关闭")

案例三：CSV 格式化写入

# 案例：用逗号分隔格式保存（方便其他程序读取）
students = [
    {"name": "张三", "id": "2024001", "phone": "13800138000"},
    {"name": "李四", "id": "2024002", "phone": "13900139000"},
    {"name": "王五", "id": "2024003", "phone": "15000150000"},
]
f = None
try:
    f = open("students.csv", "w")
    f.write("学号,姓名,手机号\n")  # CSV格式表头
    for stu in students:
        line = stu["id"] + "," + stu["name"] + "," + stu["phone"] + "\n"
        f.write(line)
    print("CSV格式的学生信息已保存！")
    print("可以用Excel或WPS直接打开students.csv查看")
except Exception as e:
    print(f"保存失败：{e}")
finally:
    if f:
        f.close()
